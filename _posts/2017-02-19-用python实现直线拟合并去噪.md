# -*- coding: utf-8 -*-
import numpy as np

def fit(pts):
    
    M = len(pts)
    x_bar=np.mean(pts[:,0])
    sum_yx= 0
    sum_x2=0
    sum_delta =0
    for i in range(M):
        x=pts[i,0]
        y=pts[i,1]
        sum_yx += y*(x-x_bar)
        sum_x2 += x**2
    #计算 k
    k = sum_yx/(sum_x2-M*(x_bar**2))
    
    for i in range(M):
        x=pts[i,0]
        y=pts[i,1] 
        sum_delta += (y-k*x)
    b = sum_delta / M

    distances = np.abs(pts[:,0] * k + b - pts[:, 1])
    mu =np.mean(distances) #计算均值 
    sigma =np.std(distances)
    index = np.where((distances <= mu + 3 * (sigma ** 2)))
    points_no_noise = pts[index]
